确保脚本执行与Excel配置完全一致
====================================================================

一、问题分析
--------------------------------------------------------------------

### 用户关注点

**核心问题**：
"怎么才能确保脚本帮我实现的内容与我的task内的文件内容是完全一致的呢？"

**担心的风险**：
1. ❌ 脚本读取Excel时出错（编码、格式问题）
2. ❌ 脚本理解错误（选错账户、链接、素材等）
3. ❌ 执行过程中遗漏步骤
4. ❌ 服务器端保存的数据与预期不符
5. ❌ 并行模式下任务串扰

**影响**：
- 广告配置错误 → 浪费预算
- 素材错位 → 投放效果差
- 数据不一致 → 难以追踪问题


二、验证方案架构
--------------------------------------------------------------------

### 多层验证机制

```
┌────────────────────────────────────────┐
│ 第1层：执行前验证（Excel数据）         │
│ - 读取数据完整性检查                   │

│ - 数据摘要对比                         │
└────────────────────────────────────────┘
              ↓
┌────────────────────────────────────────┐

```

// lib/excel.js 增强版
function validateTaskData(taskGroup) {
  const errors = [];
  
  // 检查必填字段
  if (!taskGroup.taskId) errors.push('缺少任务ID');
  if (!taskGroup.entity) errors.push('缺少主体');
  if (!taskGroup.projectName) errors.push('缺少项目名称');
  if (!taskGroup.accounts || taskGroup.accounts.length === 0) {
    errors.push('缺少账户');
  }
  
  // 检查账户数据
  taskGroup.accounts.forEach((account, index) => {
    if (!account.accountId) {
      errors.push(`账户${index+1}缺少账户ID`);
    }
    if (!account.linkKeyword) {
      errors.push(`账户${index+1}缺少推广链接`);
    }
    if (!account.titles || account.titles.length === 0) {
      errors.push(`账户${index+1}缺少标题`);
    }
  });
  
  // 检查素材检索
  if (!taskGroup.materialKeyword && 
      (!taskGroup.materialIds || taskGroup.materialIds.length === 0)) {
    errors.push('缺少素材关键词或素材ID');
  }
  
  // 检查出价和预算
  if (!taskGroup.bid || taskGroup.bid <= 0) {
    errors.push('出价无效');
  }
  if (!taskGroup.budget || taskGroup.budget < 50) {
    errors.push('预算无效（最低50）');
  }
  
  return errors;
}
```

// lib/logger.js（新文件）

class DetailedLogger {
  constructor(taskId) {
    this.taskId = taskId;
    this.logFile = `./logs/${taskId}-${Date.now()}.log`;
    this.records = [];
  }
  
  logAction(step, field, expectedValue, actualValue) {
    const record = {
      timestamp: new Date().toISOString(),
      step: step,
      field: field,
      expected: expectedValue,
      actual: actualValue,
      match: expectedValue === actualValue
    };
    
    this.records.push(record);
    
    // 实时输出
    if (record.match) {
      console.log(`  ✓ ${field}: ${actualValue}`);
    } else {
      console.log(`  ⚠️  ${field}: 预期"${expectedValue}", 实际"${actualValue}"`);
    }
    
    // 写入日志文件
    fs.appendFileSync(this.logFile, JSON.stringify(record) + '\n');
  }
  
  generateReport() {
    const mismatches = this.records.filter(r => !r.match);
    
    if (mismatches.length > 0) {
      console.log('\n⚠️  发现数据不一致：');
      mismatches.forEach(m => {
        console.log(`  - ${m.field}: 预期"${m.expected}", 实际"${m.actual}"`);
      });
    } else {
      console.log('\n✅ 所有数据与Excel一致');
    }
    
    return mismatches;
  }
}

// 使用示例
async function selectAccountsWithValidation(page, accountIds, logger) {
  log('选择账户...');
  
  for (const accountId of accountIds) {
    // 选择账户
    await selectAccount(page, accountId);
    
    // 验证是否选中
    const selectedAccounts = await page.evaluate(() => {
      return Array.from(document.querySelectorAll('.selected-account'))
        .map(el => el.textContent);
    });
    
    const isSelected = selectedAccounts.some(acc => acc.includes(accountId));
    logger.logAction('选择账户', `账户${accountId}`, '已选中', 
                     isSelected ? '已选中' : '未选中');
  }
}

async function inputBidAndBudgetWithValidation(page, expectedBid, expectedBudget, logger) {
  // 输入出价
  await bidInput.fill(String(expectedBid));
  await shortDelay();
  
  // 验证出价
  const actualBid = await bidInput.inputValue();
  logger.logAction('输入出价', '出价', expectedBid, actualBid);
  
  // 输入预算
  await budgetInput.fill(String(expectedBudget));
  await shortDelay();
  
  // 验证预算

  logger.logAction('输入预算', '预算', expectedBudget, actualBudget);
  
  // 如果不匹配，警告用户
  if (actualBid != expectedBid || actualBudget != expectedBudget) {
    log('⚠️  出价或预算输入值与预期不符！', 'WARN');
    await pauseForUser('请检查出价和预算是否正确，然后按Enter继续');

}

// lib/reporter.js（新文件）

function generateExecutionReport(taskGroups, results, logger) {
  const reportPath = `./reports/执行报告-${Date.now()}.html`;
  
  const html = `
<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <title>执行报告</title>
  <style>
    body { font-family: Arial; margin: 20px; }
    table { border-collapse: collapse; width: 100%; margin: 20px 0; }
    th, td { border: 1px solid #ddd; padding: 8px; text-align: left; }
    th { background-color: #4CAF50; color: white; }
    .match { background-color: #d4edda; }
    .mismatch { background-color: #f8d7da; }
    .success { color: green; }
    .error { color: red; }
  </style>
</head>
<body>
  <h1>🔍 执行报告</h1>
  <p>执行时间: ${new Date().toLocaleString()}</p>
  
  <h2>📋 任务概览</h2>
  <table>
    <tr>
      <th>任务ID</th>
      <th>项目名称</th>
      <th>账户数</th>
      <th>执行结果</th>
      <th>数据一致性</th>
    </tr>
    ${taskGroups.map((group, index) => {
      const result = results[index];
      const mismatches = logger.records.filter(r => !r.match && r.taskId === group.taskId);
      
      return `
        <tr>
          <td>${group.taskId}</td>
          <td>${group.projectName}</td>
          <td>${group.accounts.length}</td>
          <td class="${result.success ? 'success' : 'error'}">
            ${result.success ? '✅ 成功' : '❌ 失败'}
          </td>
          <td class="${mismatches.length === 0 ? 'match' : 'mismatch'}">
            ${mismatches.length === 0 ? '✅ 一致' : `⚠️  ${mismatches.length}处不一致`}
          </td>
        </tr>
      `;
    }).join('')}
  </table>
  
  <h2>⚠️  数据不一致详情</h2>
  ${logger.records.filter(r => !r.match).length === 0 ? 
    '<p style="color: green;">✅ 所有数据与Excel完全一致</p>' :
    `<table>
      <tr>
        <th>任务</th>
        <th>步骤</th>
        <th>字段</th>
        <th>Excel预期值</th>
        <th>实际执行值</th>
      </tr>
      ${logger.records.filter(r => !r.match).map(r => `
        <tr>
          <td>${r.taskId}</td>
          <td>${r.step}</td>
          <td>${r.field}</td>
          <td>${r.expected}</td>
          <td>${r.actual}</td>
        </tr>
      `).join('')}
    </table>`
  }
  
  <h2>📸 截图</h2>
  <p>截图保存在: ./screenshots/</p>
  
  <h2>📝 详细日志</h2>
  <p>日志文件: ${logger.logFile}</p>
</body>
</html>
  `;
  
  fs.writeFileSync(reportPath, html);
  log(`✅ 执行报告已生成: ${reportPath}`);
  
  return reportPath;
}

// 生成Excel对比文件

const ExcelJS = require('exceljs');

async function generateComparisonExcel(taskGroups, executionResults) {
  const workbook = new ExcelJS.Workbook();
  const worksheet = workbook.addWorksheet('数据对比');
  
  // 表头
  worksheet.columns = [
    { header: '任务ID', key: 'taskId', width: 15 },
    { header: '字段', key: 'field', width: 20 },
    { header: 'Excel配置', key: 'expected', width: 30 },
    { header: '实际执行', key: 'actual', width: 30 },
    { header: '是否一致', key: 'match', width: 10 },
  ];
  
  // 数据行
  executionResults.forEach(result => {
    result.validations.forEach(v => {
      const row = worksheet.addRow({
        taskId: result.taskId,
        field: v.field,
        expected: v.expected,
        actual: v.actual,
        match: v.match ? '✅' : '❌'
      });
      
      // 不一致的行标红
      if (!v.match) {
        row.getCell('match').fill = {
          type: 'pattern',
          pattern: 'solid',
          fgColor: { argb: 'FFFF0000' }
        };
      }
    });
  });
  
  await workbook.xlsx.writeFile('./reports/数据对比.xlsx');
  log('✅ 对比表格已生成: ./reports/数据对比.xlsx');
}
```
